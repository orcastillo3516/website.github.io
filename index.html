Ejemplo de Encuesta con Estrellas

En esta página puedes ver el código HTML/JS para la encuesta con estrellas. Copia y pega este código en tu propio archivo .html para probarlo en tu navegador.

Vista Previa del Código

<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <title>Encuesta Restaurante</title>
</head>
<body>
  <h1>Encuesta de Satisfacción</h1>

  <!-- Entrada de correo -->
  <label>Correo electrónico:</label>
  <input type="email" id="correo" placeholder="ejemplo@correo.com">
  <br><br>

  <!-- Calificación con estrellas (imágenes) -->
  <label>Califique el restaurante:</label><br>
  <img src="estrella_vacia.png" id="s1" onclick="calificar(1)">
  <img src="estrella_vacia.png" id="s2" onclick="calificar(2)">
  <img src="estrella_vacia.png" id="s3" onclick="calificar(3)">
  <img src="estrella_vacia.png" id="s4" onclick="calificar(4)">
  <img src="estrella_vacia.png" id="s5" onclick="calificar(5)">

  <h2>Resultados</h2>
  <ul id="listaResultados"></ul>

  <script>
    function calificar(valor) {
      let correoInput = document.getElementById("correo");
      let correo = correoInput.value.trim();

      if (correo === "") {
        alert("Por favor ingrese un correo antes de calificar.");
        return;
      }

      let li = document.createElement("li");
      li.textContent = correo + " Calificación: " + valor + " estrella(s)";
      document.getElementById("listaResultados").appendChild(li);

      correoInput.value = "";
    }
  </script>
</body>
</html>
